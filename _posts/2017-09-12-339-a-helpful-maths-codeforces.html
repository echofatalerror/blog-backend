Solución

La solución a este problema es sencilla. Almacenaremos los números que nos dan en el input en un vector. Luego ordenaremos el vector y finalmente lo mostramos. El método que uso para ordenar el vector es: *HeapSort*.

    #include 
    using namespace std;
    
    void heapify(int arr[], int n, int i) {
        int largest = i;
        int l = 2*i + 1;
        int r = 2*i + 2;
    
        if (l < n && arr[l] > arr[largest]) largest = l;
        if (r < n && arr[r] > arr[largest]) largest = r;
    
        if (largest != i) {
            swap(arr[i], arr[largest]);
            heapify(arr, n, largest);
        }
    }
    
    void heapSort(int arr[], int n) {
        for (int i = n / 2 - 1; i >= 0; i--)
            heapify(arr, n, i);
    
        for (int i=n-1; i>=0; i--) {
            swap(arr[0], arr[i]);
    
            heapify(arr, i, 0);
        }
    }
    
    int main() {
        int n[100], j = 0;
        string s;
        getline(cin, s);
    
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == '+') continue;
            else {
                n[j] = s[i] - '0';  j++;
            }
        }
    
        heapSort(n, j);
    
        for (int i = 0; i < j; i++) {
            if (i < j - 1) cout << n[i] << "+";
            else cout << n[i];
        }
    
        return 0;
    }
